10 - Closure Parameters, Results, and Methods

Closures with multiple PRE and POST checking functions must be wrapped up in objects like below for everything to be linked up. 
function PRE_check_yourClosure(init_closure){ // before closure }

function POST_check_yourClosure(result_closure){ // after closure }

function PRE_check_yourFunction(function_input){ // before function }

function POST_check_yourFunction(function_output){ // after function }

yourClosure_PRE_checks = {yourClosure:PRE_check_yourClosure,
                          yourFunction:PRE_check_yourFunction}

yourClosure_POST_checks = {yourClosure:POST_check_yourClosure,
                           yourFunction:POST_check_yourFunction}

yourClosure = type_czech.check(yourClosure, 
                               yourClosure_PRE_checks,
                               yourClosure_POST_checks)

function yourClosure(init_closure){
  yourFunction = function(function_input){ 
    return function_output
   }

  result_object = yourFunction
  return result_closure
}
